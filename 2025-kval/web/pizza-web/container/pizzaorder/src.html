import jakarta.servlet.*;
import jakarta.servlet.http.*;
import jakarta.servlet.http.HttpServletResponse;
import java.io.*;
import java.util.*;
import org.apache.commons.text.StringSubstitutor;


public class OrderServlet extends HttpServlet
{
	StringSubstitutor ipl = StringSubstitutor.createInterpolator();
	Map<String, String> valuesMap = new HashMap<String, String>();
	Map<String, String> pizzasMap = new HashMap<String, String>();
	String pizzatype = "";
	String address = "";
	String message = "<h1>You have ordered a $${pizza-${type}} to the address ${address}</h1>";
	String output = "";
	
	public void doGet(HttpServletRequest request,HttpServletResponse response) throws IOException, ServletException {
		
		
		response.setContentType("text/html");
		PrintWriter out = response.getWriter();
		
		pizzatype = request.getParameter("type");
		address = request.getParameter("address");
		valuesMap.put("type", pizzatype);
		valuesMap.put("address", address.replaceAll("\\{|\\}","|"));
		pizzasMap.put("pizza-pineapple", "pizza with pineapple");
		pizzasMap.put("pizza-pepperoni", "pepperonipizza");
		pizzasMap.put("pizza-funghi", "pizza med funghi");
		pizzasMap.put("pizza-cheese", "pizza with extra cheese");
		
		output = ipl.replace(ipl.replace(message,valuesMap),pizzasMap);
		
		//todo: add saving to file
		
		out.println("<html><head><title>Pizza Web</title></head><body>");
		out.println(ipl.replace(output));
		out.println("<p>You will soon be sent to another page. This is just a page to register your order.</p><script src='index.js'></script><script>finishOrder()</script></body></html>");
	}
}
